require('dotenv').config(); const { Client, GatewayIntentBits, Collection } = require('discord.js'); const fs = require('fs'); const path = require('path'); const { QuickDB } = require('quick.db'); const express = require('express'); const cors = require('cors'); const app = express(); app.use(express.urlencoded({ extended: true })); app.use(express.json()); app.use(cors()); const client = new Client({ intents: [ GatewayIntentBits.Guilds, GatewayIntentBits.GuildMembers, GatewayIntentBits.GuildMessages, GatewayIntentBits.MessageContent, ], }); client.commands = new Collection(); client.db = new QuickDB(); const commandsPath = path.join(__dirname, 'commands'); const commandFiles = fs.readdirSync(commandsPath).filter(file => file.endsWith('.js')); for (const file of commandFiles) { const filePath = path.join(commandsPath, file); const command = require(filePath); if ('data' in command && 'execute' in command) { client.commands.set(command.data.name, command); } else { console.log(`[WARNING] The command at ${filePath} is missing a required "data" or "execute" property.`); } } client.once('ready', () => { console.log(`✅ Bot aktif! Giriş yapıldı: ${client.user.tag}`); app.listen(3000, () => { console.log('🌐 Web sunucusu 3000 portunda çalışıyor'); console.log('🔐 Kod doğrulama API: http://localhost:3000/verify-code'); }); }); client.on('messageCreate', async message => { if (message.author.bot) return; const content = message.content.toLowerCase(); if (content === 'sa' || content === 's a' || content === 'sA' || content === 'Sa') { await message.reply('as'); } }); client.on('interactionCreate', async interaction => { if (interaction.isChatInputCommand()) { const command = client.commands.get(interaction.commandName); if (!command) return; try { await command.execute(interaction); } catch (error) { console.error(error); await interaction.reply({ content: 'There was an error while executing this command!', ephemeral: true }); } } else if (interaction.isModalSubmit()) { if (interaction.customId === 'alim_denetim_modal' || interaction.customId === 'egitim_denetim_modal') { const not = interaction.fields.getTextInputValue('not'); const durum = interaction.fields.getTextInputValue('durum'); // Webhook ile gönder, şimdilik kanal mesajı const channel = interaction.guild.channels.cache.find(ch => ch.name === 'log-kanali'); // Varsay if (channel) { await channel.send(`**${interaction.customId} Raporu**\nNot: ${not}\nDurum: ${durum}\nGönderen: ${interaction.user.tag}`); } await interaction.reply({ content: 'Rapor gönderildi!', ephemeral: true }); } else if (interaction.customId === 'hata_rapor_modal') { const hata = interaction.fields.getTextInputValue('hata'); const channel = interaction.guild.channels.cache.find(ch => ch.name === 'log-kanali'); if (channel) { await channel.send(`**Hata Raporu**\nHata: ${hata}\nGönderen: ${interaction.user.tag}`); } await interaction.reply({ content: 'Hata raporu gönderildi!', ephemeral: true }); } } }); app.get('/', (req, res) => { res.send(` Giriş

Giriş Yapın

`); }); app.post('/giris', async (req, res) => { const kod = req.body.kod.toUpperCase(); const valid = await client.db.get(`kod_${kod}`); if (valid) { res.send(` Ana Sayfa

Hoş Geldiniz!

Gizli içerik burada: Bu botun özel sayfası.

`); } else { res.send(` Hata

Giriş Başarısız

Geçersiz kod.

`); } }); // ===== SİTEDEN GELEN KOD DOĞRULAMA API'Sİ ===== app.post('/verify-code', async (req, res) => { try { const { code } = req.body; // Kod boşsa hata dön if (!code || typeof code !== 'string') { return res.status(400).json({ valid: false, message: 'Kod belirtilmedi' }); } // Kodu normalize et (büyük harfe çevir ve boşlukları kaldır) const normalizedCode = code.trim().toUpperCase(); // Veritabanında kodu ara const codeData = await client.db.get(`kod_${normalizedCode}`); if (codeData && codeData.valid) { // Kod geçerli console.log(`✅ Kod doğrulandı: ${normalizedCode}`); // Log kaydet const logEntry = { userId: codeData.userId, username: codeData.username, discriminator: codeData.discriminator, code: normalizedCode, loginTime: new Date().toISOString(), ip: req.ip || req.connection.remoteAddress, logoutTime: null }; await client.db.push('logs', logEntry); return res.json({ valid: true, message: 'Kod doğrulandı', user: { id: codeData.userId, username: codeData.username, discriminator: codeData.discriminator, avatar: codeData.avatar } }); } else { // Kod bulunamadı veya geçersiz console.log(`❌ Geçersiz kod girişi: ${normalizedCode}`); return res.status(401).json({ valid: false, message: '❌ Geçersiz kod!' }); } } catch (error) { console.error('Kod doğrulama hatası:', error); return res.status(500).json({ valid: false, message: 'Sunucu hatası' }); } }); app.get('/api/logs', async (req, res) => { try { const logs = await client.db.get('logs') || []; res.json(logs); } catch (error) { console.error('Log getirme hatası:', error); res.status(500).json({ error: 'Sunucu hatası' }); } }); // Botu Discord'a giriş yap client.login(process.env.DISCORD_TOKEN).catch(err => { console.error('Bot login hatası:', err); });
